nota = float(input("Introduce la nota: "))  

if 0 <= nota < 5:
    print("suspenso")
elif 5 <= nota < 6:
    print("suficiente")
elif 6 <= nota < 7:
    print("bien")
elif 7 <= nota < 9:
    print("notable")
elif 9 <= nota <= 10:
    print("sobresaliente")
else:
    print("NOTA NO VÃLIDA")
